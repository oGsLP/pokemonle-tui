```python
def _hint_signature(hints: list[Hint]) -> tuple[Hint, ...]:
    return tuple(hints)


def compute_remaining_pool(pool: list[dict], guesses_with_hints: list[tuple[dict, list[Hint]]], config: dict) -> list[dict]:
    """Return candidates that would produce the same hints for all previous guesses."""
    remaining = []
    for candidate in pool:
        candidate_matches = True
        for guess_poke, observed_hints in guesses_with_hints:
            simulated_hints = compare_pokemon(candidate, guess_poke, config)
            if _hint_signature(simulated_hints) != _hint_signature(observed_hints):
                candidate_matches = False
                break
        if candidate_matches:
            remaining.append(candidate)
    return remaining
```
